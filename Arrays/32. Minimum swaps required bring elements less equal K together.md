## Code :

```python
import sys
nums = [2, 1, 5, 6, 3]
k = 5

def fun(nums,k):
   lessThanK = 0
   greaterThanK = 0
   minSwaps = sys.maxsize
   
   for num in nums:
      if num <= k:
         lessThanK += 1

   for i in range(0,k-1):
      if nums[i] > k:
         greaterThanK += 1

   # Sliding Window
   i = 0
   j = lessThanK - 1
   minSwaps = greaterThanK

   while j < len(nums):

      if nums[i] > k:
         greaterThanK -=1
      if nums[j] > k:
         greaterThanK +=1

      minSwaps = min(minSwaps,greaterThanK)

      i += 1
      j += 1

   return minSwaps

print(fun(nums,k))
```
